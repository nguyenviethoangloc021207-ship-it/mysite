(function() {
    const mainMusic = document.getElementById('backgroundMusic');
    const seawaveSound = document.getElementById('seawaves');

    function tryPlayAllMusic() {
        // Cố gắng chạy nhạc nền chính
        if (mainMusic) {
            mainMusic.play().catch(error => {
                console.log("Không thể chạy nhạc chính.");
            });
        }
        
        // Cố gắng chạy âm thanh hiệu ứng (sóng biển)
        if (seawaveSound) {
            seawaveSound.play().catch(error => {
                console.log("Không thể chạy âm thanh sóng biển.");
            });
        }

        // Sau lần tương tác đầu tiên, gỡ bỏ listener
        if (mainMusic || seawaveSound) {
            document.removeEventListener('click', tryPlayAllMusic);
            document.removeEventListener('touchstart', tryPlayAllMusic);
            console.log("Cả hai âm thanh đã được kích hoạt.");
        }
    }

    // Lắng nghe sự kiện tương tác của người dùng
    document.addEventListener('click', tryPlayAllMusic);
    document.addEventListener('touchstart', tryPlayAllMusic);
})();

(function() {
  let currentIndex = 0;
  const slider = document.getElementById('sliderItems');
  const items = document.querySelectorAll('.slide-item');
  const totalItems = items.length;
  let itemsPerView = getItemsPerView();
  let maxIndex = calculateMaxIndex();

  function getItemsPerView() {

    if (window.innerWidth <= 480) {
      return 1; 
    } else if (window.innerWidth <= 768) {
      return 3; 
    } else if (window.innerWidth <= 992) {
      return 3;
    } else if (window.innerWidth <= 1200) {
      return 4;
    } else if (window.innerWidth <= 1600) {
      return 5;
    } else {

      return 6; 
    }
  }

  function calculateMaxIndex() {
    itemsPerView = getItemsPerView(); 
    return totalItems - itemsPerView;
  }

  function updateSlider() {
    maxIndex = calculateMaxIndex();
    
    if (currentIndex > maxIndex) {
      currentIndex = maxIndex;
    }
    if (currentIndex < 0) {
      currentIndex = 0;
    }
    
    const itemWidth = items[0].offsetWidth; 
    slider.style.transform = 'translateX(-' + (currentIndex * itemWidth) + 'px)';
  }

  document.getElementById('prevBtn').addEventListener('click', function() {
    if (currentIndex > 0) {
      currentIndex--;
    } else {
      currentIndex = maxIndex; 
    }
    updateSlider();
  });

  document.getElementById('nextBtn').addEventListener('click', function() {
    if (currentIndex < maxIndex) {
      currentIndex++;
    } else {
      currentIndex = 0;
    }
    updateSlider();
  });

  // Tự động trượt (Auto slide) - Đã tắt theo yêu cầu người dùng
  // setInterval(function() {
  //   if (currentIndex < maxIndex) {
  //     currentIndex++;
  //   } else {
  //     currentIndex = 0;
  //   }
  //   updateSlider();
  // }, 4000);

  // Cập nhật khi thay đổi kích thước cửa sổ
  let resizeTimer;
  window.addEventListener('resize', function() {
    clearTimeout(resizeTimer);
    resizeTimer = setTimeout(updateSlider, 200);
  });
  
  // Gọi 1 lần lúc tải trang
  updateSlider();
})();
</script>
